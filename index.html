    <button onclick="createCommitments()">Create Commitments</button>

    <script>

        var selectedAccountIds = [];


        ZOHO.embeddedApp.on("PageLoad", function(data) {

            selectedAccountIds = data.EntityId;

            if (!Array.isArray(selectedAccountIds)) {

                selectedAccountIds = [selectedAccountIds];

            }

            document.getElementById("accountCount").innerText = 

                "Selected " + selectedAccountIds.length + " Account(s). Choose an Investment:";

            loadInvestments();

        });


        ZOHO.embeddedApp.init();


        function loadInvestments() {

            ZOHO.CRM.API.getAllRecords({

                Entity: "Investments",

                sort_order: "asc",

                per_page: 200

            }).then(function(data) {

                var dropdown = document.getElementById("investmentDropdown");

                dropdown.innerHTML = '<option value="">-- Select an Investment --</option>';

                if (data.data) {

                    data.data.forEach(function(inv) {

                        var option = document.createElement("option");

                        option.value = inv.id;

                        option.text = inv.Name;

                        dropdown.appendChild(option);

                    });

                } else {

                    document.getElementById("error").innerText = "No Investments found.";

                }

            });

        }


        function createCommitments() {

            var dropdown = document.getElementById("investmentDropdown");

            var investment_id = dropdown.value;

            var investment_name = dropdown.options[dropdown.selectedIndex].text;


            if (!investment_id) {

                document.getElementById("error").innerText = "Please select an Investment.";

                return;

            }


            document.getElementById("error").innerText = "";

            document.getElementById("status").innerText = "Creating Commitments...";


            var account_ids = selectedAccountIds.join(",");


            ZOHO.CRM.FUNCTIONS.execute("createcommitmentsfromaccounts", {

                arguments: JSON.stringify({

                    investment_id: investment_id,

                    investment_name: investment_name,

                    account_ids: account_ids

                })

            }).then(function(response) {

                document.getElementById("status").innerText = "Commitments created successfully!";

            }).catch(function(err) {

                document.getElementById("error").innerText = "Error: " + JSON.stringify(err);

                document.getElementById("status").innerText = "";

            });

        }

    </script>
